 \begin{Zutaten}

		\item 50g Honig
		\item 100g Schokolade
		\item 70g weiche Butter
		\item 100ml Vollmilch
		
		
\end{Zutaten}

	Kokosbutter, Milchersatz, verschiedene Schokoladen, Kaschewmus usw.
