\chapter*{Warmer Krautsalat ala Gabi}
\begin{multicols}{2}
 {\Large Zutaten}
 \begin{Zutaten}
		\item \sfac{1}{2} Häupl Kraut
		\item Gewürfelte Bauchspeck
		\item Schweinsfett
		\item 2-3 Löffel Zucker
		\item 1 Schuss Essig
		\item 1 Prise Kümmel
		\item Pfeffer und Salz
				
		
\end{Zutaten}

{\Large Zubereitung} \newline
\begin{addmargin}[1cm]{0cm}
	Kraut hobeln oder in Streifen schneiden, größe nach eigenem Ermessen.\newline
	Bauchspeck anbraten und zur Seite tun.\newline
	Fett heiß machen und Zucker darin karamellisieren lassen.\newline
	Mit Essig ablöschen und Kraut dazu.\newline
	Mit Wasser fast leicht bedecken, Kümmel dazu und dann bei geschlossenem Deckel ca \sfac{3}{4}h weich kochen
	lassen.\newline
	Speckwürfel dazu und abschmecken.\newline
	
	
	
\end{addmargin}
